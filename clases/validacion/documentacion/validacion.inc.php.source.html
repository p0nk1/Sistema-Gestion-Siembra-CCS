   2  /**

   3  * Validación de Formularios

   4  * 

   5  * Valida formularios vía JavaScript y PHP. 

   6  * 

   7  * Instrucciones:

   8  * Para validar un formulario mediante JavaScript y PHP se debe cumplir unos pasos:

   9  * 1) Incluir el archivo que contiene la clase:

  10  * ej. include "validacion/validacion.inc.php";

  11  * 

  12  * 2) Inicializar el objeto. Se le pasa como parámetro el nombre del formulario a validar:

  13  * $validar = new MCI_validacion("login");

  14  * 

  15  * 3) Agregar un campo a la clase. El primer parámetro es el nombre del campo de formulario,

  16  * el segundo es el tipo de campo (text, textarea, select, radio, etc.), y el tercero es el Título

  17  * del campo (el cual se utiliza para hacerle referencia en el momento de tener que mostrar un error):

  18  * $validar->agregar_campo("user", "text", "Nombre de Usuario");

  19  * 

  20  * 4) Agregar uno o varios tipos de validación para el campo. El primer parámetro es el nombre del campo,

  21  * el segundo es el tipo de validación (la lista de todos los tipos de validación posibles está guardado

  22  * en la variable $this->lista_validacion). El tercer parámetro son las opciones o parámetros que se le

  23  * pasa al tipo de validación, como por ejemplo un valor en KB mayor el cual será el límite para una 

  24  * validación de tamaño de archivo. El último parámetro es un mensaje de error que suplantará al predeteminado.

  25  * Los parámetros 2 y 3 son opcionales.

  26  * $validar->agregar_validacion("user", "req", NULL, "No metiste datos al campo Usuario");

  27  * $validar->agregar_validacion("user", "a1");

  28  * $validar->agregar_validacion("user", "! ");

  29  * 

  30  * 5) Se repiten los pasos 3 y 4 tantas veces como sea necesario para agregar todos los campos y sus validaciones.

  31  * 

  32  * 6) Procesar el formulario en caso de que haya sido enviado. Devuelve verdadero si todos los campos son válidos

  33  * por JavaScript y PHP. Este es el método clave, se le puede utilizar para pasar al proximo paso en el 

  34  * procesado de datos, como por ejemplo su carga a una base de datos.

  35  * if($validar->procesar()){die("Formulario completado exitosamente!!");}

  36  * 

  37  * Los próximos pasos son de construcción del html y código JavaScript necesarios para la validación por JavaScript

  38  * 

  39  * 7) Incluir  los archivos javascript con las clases de validación. En el caso de no encontrarse en la ubicación 

  40  * predeterminada, se le puede pasar la nueva ubicación como parámetro. Estos archivos se deben incluír antes del 

  41  * formulario, preferiblemente en la sección HEAD del html.

  42  * <? echo $validar->obtener_archivos_js("validacion/js/"); ?>

  43  * 

  44  * 8) Incluír el texto de error. Este texto cambiará dinámicamente de acuerdo al error en la validación que se presente.

  45  * En el caso de haber varios formularios y que se quiera tener igual número de mensajes de error, se puede pasar

  46  * el nombre del formulario al cual estará asociado como parámetro.

  47  * <?php echo  $validar->html_error("login"); ?>

  48  * 

  49  * 9) Incluír el campo oculto. Se utiliza este método para crear un campo oculto que le indica al método procesar si el 

  50  * formulario ha sido enviado por el usuario o no, para así determinar si debe realizar la validación por PHP.

  51  * Se debe llamar este método desde adentro del formulario. Es decir, entre los campos <form> y </form>.

  52  * <?php echo  $validar->campo_oculto(); ?>

  53  * 

  54  * 10) Incluír JavaScript de validación. Es importante que se coloque este llamado después del formulario, es decir

  55  * después del tag </form> para que funcione correctamente. Esto se debe a que el JavaScript de validación trabaja 

  56  * con objetos y debe estar completo el objeto del formulario antes de poderle hacer referencia.

  57  * <?php echo $validar->obtener_js(); ?>

  58  *

  59  * @Autor Thomas Woodard

  60  */

 151      /**

 152      * Procesar

 153      * 

 154      * Se llama esta función para determinar si ya se recibieron los datos enviados por el usuario y

 158  	function procesar($usar_session=0){

 159          if ($usar_session == 1){//todavía no se puede usar esta opción

 160              // Esta código debe ser después de todos los cambios que se le hagan al objeto

 161              @session_start();

 162              if($_SESSION["MCI_validacion"][$this->nom_form]->validacion_paso != 1){

 163                  $this->validacion_paso = 1;

 164                  $_SESSION["MCI_validacion"][$this->nom_form] = $this;

 165              } else {}

 166          }else{

 167              switch($this->metodo_form){

 168                  case "POST":

 169                      if (isset($_POST["MCI_validar"]) && $_POST["MCI_validar"] == $this->nom_form){

 170                          foreach($_POST as $campo => $valor){

 171                              $this->campos[$campo]["valor"] = $valor;

 172                              $this->validacion_paso = 2;

 173                          }

 174                      }

 175                      break;

 176                  case "GET":

 177                      if (isset($_GET["MCI_validar"]) && $_GET["MCI_validar"]==$this->nom_form){

 178                          foreach($_GET as $campo => $valor){

 179                              $this->campos[$campo]["valor"] = $valor;

 180                              $this->validacion_paso = 2;

 181                          }

 182                      }

 183                      break;

 184              }

 185              if (isset($_POST["MCI_validar"]) && $_POST["MCI_validar"] == $this->nom_form){

 186                          foreach($_FILES as $campo => $campo_datos_archivo){

 187                              $this->campos[$campo]["valor"] = $campo_datos_archivo['name'];

 188                              $this->campos[$campo]["archivo"] = $campo_datos_archivo;

 189                              $this->validacion_paso = 2;

 190                          }

 191              }

 192          }

 193          if($this->validacion_paso == 2){return $this->_validacion_php();}

 194      }

 334  	function _validacion_php(){

 335          foreach ($this->campos as $id_campo => $campo){ // cada campo

 336              foreach($campo["val"] as $clave => $val){ // cada validación del campo

 337                  if($campo["titulo"]!=NULL){$nombre = $campo["titulo"];}else{$nombre = $id_campo;}

 338                  switch ($val){

 339                      case "r":

 340                      case "req":

 341                          if(!isset($campo["valor"]) || $campo["valor"] == ""){

 342                              if($campo["error"][$clave]){

 343                                  $this->validacion_error = $campo["error"][$clave];

 344                              } else $this->validacion_error = $nombre. " es un campo requerido"; 

 345                              $this->validacion_campo_error = $id_campo;

 346                              return false;

 347                          }

 348                          break;

 349                      case "maxl": 

 350                      case "maxlon": 

 351                          if(isset($campo["valor"]) && $campo["valor"] != ""){ // se verifica que se haya ingresado un valor para evitar errores

 352                              if(strlen($campo["valor"]) > $campo["val_args"][$clave]){ 

 353                                  if($campo["error"][$clave]){

 354                                      $this->validacion_error = $campo["error"][$clave];

 355                                  } else $this->validacion_error = $nombre. " debe tener no más de ".$campo["val_args"][$clave]." caracteres "; 

 356                                  $this->validacion_campo_error = $id_campo;

 357                                  return false;

 358                              }

 359                          }                        

 360                          break;

 361                      case "minl":

 362                      case "minlon":

 363                          if(isset($campo["valor"]) && $campo["valor"] != ""){

 364                              if(strlen($campo["valor"]) < $campo["val_args"][$clave]){ 

 365                                  if($campo["error"][$clave]){

 366                                      $this->validacion_error = $campo["error"][$clave];

 367                                  } else $this->validacion_error = $nombre. " debe tener almenos ".$campo["val_args"][$clave]." caracteres "; 

 368                                  $this->validacion_campo_error = $id_campo;

 369                                  $this->validacion_campo_error = $id_campo;

 370                                  return false;

 371                              }

 372                          }

 373                          break;

 374                      case "noespacio": 

 375                      case "! ":

 376                          if(isset($campo["valor"]) && $campo["valor"] != ""){

 377                              if(ereg_replace(" ", "", $campo["valor"]) != $campo["valor"]){

 378                                  if($campo["error"][$clave]){

 379                                      $this->validacion_error = $campo["error"][$clave];

 380                                  } else $this->validacion_error = $nombre. " no debe poseer espacios en blanco "; 

 381                                  $this->validacion_campo_error = $id_campo;

 382                                  return false;

 383                              }

 384                          }

 385                          break;

 386                      case "a1":

 387                      case "alfanumerico":

 388                          if(isset($campo["valor"]) && $campo["valor"] != ""){

 389                              if(ereg_replace("[^a-zA-z0-9 ]", "", $campo["valor"]) != $campo["valor"]){

 390                                  if($campo["error"][$clave]){

 391                                      $this->validacion_error = $campo["error"][$clave];

 392                                  } else $this->validacion_error = $nombre. " solamente debe tener caracteres alfa-numéricos "; 

 393                                  $this->validacion_campo_error = $id_campo;

 394                                  return false;

 395                              }

 396                          }

 397                          break;

 398                      case "1":

 399                      case "num":

 400                      case "numerico": 

 401                          if(isset($campo["valor"]) && $campo["valor"] != ""){

 402                              if(ereg_replace("[^0-9 ]", "", $campo["valor"]) != $campo["valor"]){

 403                                  if($campo["error"][$clave]){

 404                                      $this->validacion_error = $campo["error"][$clave];

 405                                  } else $this->validacion_error = $nombre. " solamente debe tener caracteres numéricos "; 

 406                                  $this->validacion_campo_error = $id_campo;

 407                                  return false;

 408                              }

 409                          }

 410                          break;

 411                      case "a":

 412                      case "alfa":

 413                      case "alfabetico":

 414                          if(isset($campo["valor"]) && $campo["valor"] != ""){

 415                              if(ereg_replace("[^a-zA-z ]", "", $campo["valor"]) != $campo["valor"]){

 416                                  if($campo["error"][$clave]){

 417                                      $this->validacion_error = $campo["error"][$clave];

 418                                  } else $this->validacion_error = $nombre. " solamente debe tener caracteres alfabéticos "; 

 419                                  $this->validacion_campo_error = $id_campo;

 420                                  return false;

 422                          }

 423                          break;

 424                      case "a1-":

 425                      case "a1-_":

 426                      case "alnumguion":

 427                          if(isset($campo["valor"]) && $campo["valor"] != ""){

 428                              if(ereg_replace("[^a-zA-z0-9._- ]", "", $campo["valor"]) != $campo["valor"]){

 429                                  if($campo["error"][$clave]){

 430                                      $this->validacion_error = $campo["error"][$clave];

 431                                  } else $this->validacion_error = $nombre. " debe incluir solamente los caracteres A-Z,a-z,0-9,- y _"; 

 432                                  $this->validacion_campo_error = $id_campo;

 433                                  return false;

 434                              }

 435                          }

 436                          break;

 437                      case "email": 

 438                          if(isset($campo["valor"]) && $campo["valor"] != ""){

 439                              $pattern = "/^[A-z0-9_\-]+\@(A-z0-9_-]+\.)+[A-z]{2,4}$/";

 440                              if(!preg_match($pattern,$campo["valor"])){

 441                                  if($campo["error"][$clave]){

 442                                      $this->validacion_error = $campo["error"][$clave];

 443                                  } else $this->validacion_error = "Debe ingresar un email válido ";

 444                              }

 445                              if (strlen($this->email)>100){

 446                                  if($campo["error"][$clave]){

 447                                      $this->validacion_error = $campo["error"][$clave];

 448                                  } else $this->validacion_error = "El correo electrónico es demasiado largo";

 449                              }

 450                              

 451                          }

 452                          break;

 453                      case "<":

 454                      case "menorque":
